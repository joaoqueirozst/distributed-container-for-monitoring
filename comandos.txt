# Em cada VM:
git clone https://github.com/joaoqueirozst/distributed-container-for-monitoring
cd distributed-container-for-monitoring
docker build -t host ./host
docker run -d --network host --name host host
docker logs -f host

docker kill host && docker rm host
rm -rf distributed-container-for-monitoring
docker rmi host

# Switches 1
git clone https://github.com/joaoqueirozst/distributed-container-for-monitoring
scp -r ~/distributed-container-for-monitoring root@'ip_sw1':~ # Fazer isso em VM1
cd distributed-container-for-monitoring
docker build -t rede ./rede
docker run -d --network host --privileged --name rede -e SWITCH=sw1 rede
docker logs -f rede
